function autoSelectSheet(pw,ph){


  if(!pw||!ph) return;


  const ew=pw+BLEED*2,eh=ph+BLEED*2;


 


  // Auto-select pouze ze SRA3 a SRA3 MAX — pick best utilization


  const AUTO_SHEETS = ['SRA3', 'SRA3 MAX'];


  const scored = SHEETS.filter(s=>AUTO_SHEETS.includes(s.name)).map(s=>({


    ...s,


    count: Math.max(


      Math.floor(s.w/ew)*Math.floor(s.h/eh),


      Math.floor(s.w/eh)*Math.floor(s.h/ew)


    )


  })).filter(s=>s.count>0)


     .map(s=>({...s, util:(s.count*ew*eh)/(s.w*s.h), area:s.w*s.h}))


     .sort((a,b)=>a.area-b.area);


 


  if(scored.length===0) return;


 


  // Best utilization; if smaller sheet within 85%, prefer smaller


  const bestUtil = scored.reduce((a,b)=>b.util>a.util?b:a).util;


  const candidate = scored.find(s=>s.util>=bestUtil*0.85);


  const best = candidate || scored[0];


  const opts=document.querySelectorAll('.sheet-option');


  let el=null;


  opts.forEach(o=>{if(parseInt(o.dataset.w)===best.w&&parseInt(o.dataset.h)===best.h)el=o;});


  if(!el) return;


  opts.forEach(o=>o.classList.remove('active','auto-selected'));


  el.classList.add('active','auto-selected');


  sheetW=best.w;sheetH=best.h;sheetName=best.name;


  const note=document.getElementById('autoSheetNote');


  note.innerHTML='✅ Automaticky vybrán <strong>'+best.name+'</strong> ('+best.w+' × '+best.h+' mm)';


  note.classList.add('visible');


  setTimeout(()=>el.classList.remove('auto-selected'),3000);


}
